> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `setLightTheme()` and `setDarkTheme()` in `src/ui/theme.ts` so callers can switch the global `html` class between mutually exclusive `light` and `dark` modes.

**Architecture:** Keep the implementation minimal and browser-first: each exported function reads `document.documentElement` and updates `classList` directly. Lock the behavior with a focused Bun test file that installs a tiny fake `document` and verifies the two classes remain mutually exclusive.

**Tech Stack:** TypeScript, Bun test

- [ ] **Step 1: Create the focused theme test file**

Write `test/theme.test.ts` with this content:

```ts
import { afterEach, describe, expect, test } from "bun:test";

import { setDarkTheme, setLightTheme } from "../src/ui/theme";

class FakeClassList {
  private readonly values = new Set<string>();

  add(...tokens: string[]) {
    for (const token of tokens) {
      this.values.add(token);
    }
  }

  remove(...tokens: string[]) {
    for (const token of tokens) {
      this.values.delete(token);
    }
  }

  contains(token: string) {
    return this.values.has(token);
  }

  toArray() {
    return [...this.values].sort();
  }
}

const originalDocument = (globalThis as Record<string, unknown>).document;

const installDocument = (initialClasses: string[] = []) => {
  const classList = new FakeClassList();
  classList.add(...initialClasses);

  Object.defineProperty(globalThis, "document", {
    configurable: true,
    value: {
      documentElement: {
        classList,
      },
    },
  });

  return classList;
};

const restoreDocument = () => {
  if (typeof originalDocument === "undefined") {
    delete (globalThis as Record<string, unknown>).document;
    return;
  }

  Object.defineProperty(globalThis, "document", {
    configurable: true,
    value: originalDocument,
  });
};

afterEach(() => {
  restoreDocument();
});

describe("theme api", () => {
  test("setLightTheme adds light and removes dark", () => {
    const classList = installDocument(["dark"]);

    setLightTheme();

    expect(classList.contains("light")).toBe(true);
    expect(classList.contains("dark")).toBe(false);
    expect(classList.toArray()).toEqual(["light"]);
  });

  test("setDarkTheme adds dark and removes light", () => {
    const classList = installDocument(["light"]);

    setDarkTheme();

    expect(classList.contains("dark")).toBe(true);
    expect(classList.contains("light")).toBe(false);
    expect(classList.toArray()).toEqual(["dark"]);
  });

  test("theme setters keep light and dark mutually exclusive across repeated calls", () => {
    const classList = installDocument();

    setDarkTheme();
    setLightTheme();
    setDarkTheme();

    expect(classList.toArray()).toEqual(["dark"]);
  });
});
```

Write `src/ui/theme.ts` with this content:

```ts
export function setLightTheme() {
  const root = document.documentElement;
  root.classList.remove("dark");
  root.classList.add("light");
}

export function setDarkTheme() {
  const root = document.documentElement;
  root.classList.remove("light");
  root.classList.add("dark");
}
```
